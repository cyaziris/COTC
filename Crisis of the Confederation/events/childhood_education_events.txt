	max_age = 16

		name = "EVTOPTACHILDHOODEDUCATION0"

	}
}

character_event = {
	id = childhood_education.11
	desc = "EVTDESCCHILDHOODEDUCATION11"
	picture = GFX_evt_child
	is_triggered_only = yes
	
	option = {
		name = "EVTOPTACHILDHOODEDUCATION11" # Encourage behavior
		FROM = {
			add_trait = open_minded
			hidden_tooltip = { character_event = { id = childhood_education.12 } }
		}
		ai_chance = {
			factor = 50
			modifier = {
				factor = 2
				trait = skeptical
			}
			modifier = {
				factor = 0.5
				trait = spiritual
			}
			modifier = {
				factor = 0
				trait = close_minded
			}
		}
	}
	option = {
		name = "EVTOPTBCHILDHOODEDUCATION11" # Try to persuade otherwise
		if = {
			limit = { FROM = { opinion = { who = ROOT value = 0 } } }
			FROM = {
				random_list = {
					60 = {
						add_trait = close_minded
						hidden_tooltip = { character_event = { id = childhood_education.13 } }
					}
					30 = {
						remove_trait = open_minded
						hidden_tooltip = { character_event = { id = childhood_education.14 } }
					}
					10 = { hidden_tooltip = { character_event = { id = childhood_education.15 } } }
				}
			}
		}
		if = {
			limit = { FROM = { NOT = { opinion = { who = ROOT value = 0 } } } }
			FROM = {
				random_list = {
					20 = {
						add_trait = close_minded
						hidden_tooltip = { character_event = { id = childhood_education.13 } }
					}
					30 = {
						remove_trait = open_minded
						hidden_tooltip = { character_event = { id = childhood_education.14 } }
					}
					50 = { hidden_tooltip = { character_event = { id = childhood_education.15 } } }
				}
			}
		}
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0
				trait = open_minded
			}
		}
	}
	option = {
		name = "EVTOPTCCHILDHOODEDUCATION11" # Punish for misbehavior
		FROM = {
			random_list = {
				40 = {
					add_trait = close_minded
					hidden_tooltip = { character_event = { id = childhood_education.16 } }
				}
				60 = { hidden_tooltip = { character_event = { id = childhood_education.17 } } }
			}
			random = {
				chance = 10
				add_trait = neurotic
				hidden_tooltip = { character_event = { id = childhood_education.0 } }
			}
		}
		ai_chance = {
			factor = 10
			modifier = {
				factor = 3
				trait = cold
			}
			modifier = {
				factor = 3
				trait = neurotic
			}
			modifier = {
				factor = 0
				trait = friendly
			}
			modifier = {
				factor = 0
				trait = open_minded
			}
			modifier = {
				factor = 0
				NOT = { trait = neurotic }
				NOT = { trait = close_minded }
			}
		}
	}
}

character_event = {
	id = childhood_education.170
	desc = "EVTDESCCHILDHOODEDUCATION170"
	picture = GFX_evt_child
	is_triggered_only = yes
	
	min_age = 6
	max_age = 16
	
	trigger = {
		NOT = { trait = brave }
		NOT = { trait = craven }
	}
	
	weight_multiplier = {
		days = 1
		modifier = {
			factor = 2
			guardian = { trait = brave }
		}
		modifier = {
			factor = 2
			father = { trait = brave }
		}
		modifier = {
			factor = 2
			mother = { trait = brave }
		}
		modifier = {
			factor = 2
			any_sibling = { trait = brave }
		}
		modifier = {
			factor = 2
			any_friend = { trait = brave }
		}
		modifier = {
			factor = 0.5
			guardian = { trait = craven }
		}
		modifier = {
			factor = 0.5
			father = { trait = craven }
		}
		modifier = {
			factor = 0.5
			mother = { trait = craven }
		}
		modifier = {
			factor = 0.5
			any_sibling = { trait = craven }
		}
		modifier = {
			factor = 0.5
			any_friend = { trait = craven }
		}
		
		modifier = {
			factor = 5
			trait = confident
		}
		modifier = {
			factor = 0.2
			trait = neurotic
		}
		
		modifier = {
			factor = 2
			trait = open_minded
		}
		modifier = {
			factor = 0.5
			trait = close_minded
		}
	}
	
	option = {
		trigger = { has_guardian = no }
		name = "EVTOPTACHILDHOODEDUCATION170"
		add_trait = brave
	}
	option = {
		trigger = { has_guardian = yes }
		name = "EVTOPTBCHILDHOODEDUCATION170"
		add_trait = brave
		guardian = { character_event = { id = childhood_education.171 tooltip = "EVTTOOLTIPCHILDHOODEDUCATION171" days = 5 } }
	}
}

	id = childhood_education.171
	desc = "EVTDESCCHILDHOODEDUCATION171"
	picture = GFX_evt_child
	is_triggered_only = yes
	
	option = {
		name = "EVTOPTACHILDHOODEDUCATION171" # Encourage behavior
		FROM = {
			add_trait = brave
			hidden_tooltip = { character_event = { id = childhood_education.172 } }
		}
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0.5
				trait = craven
			}
			modifier = {
				factor = 0.1
				OR = {
					trait = craven
					trait = neurotic
				}
				is_close_relative = FROM
			}
			modifier = {
				factor = 0.5
				OR = {
					trait = craven
					trait = neurotic
				}
				FROM = {
					OR = {
						is_close_relative = ROOT
						trait = wounded
						trait = scarred
						trait = maimed
					}
				}
			}
		}
	}

		name = "EVTOPTBCHILDHOODEDUCATION171" # Try to persuade otherwise

					}
					10 = { hidden_tooltip = { character_event = { id = childhood_education.175 } } }
				}
			}
		}

				factor = 3

			}

				factor = 3
				trait = neurotic

				factor = 0
				trait = friendly
